async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(WELCOME_TEXT, parse_mode="Markdown", disable_web_page_preview=True)

async def user_to_admin(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.message.chat.type == "private":
        user = update.message.from_user
        admin_text = (f"📩 Новое сообщение от @{user.username or 'без ника'} (ID: {user.id})\nТекст: {update.message.text}")
        sent_msg = await context.bot.send_message(chat_id=CHAT_ID, text=admin_text)
        context.bot_data[sent_msg.message_id] = user.id
        await update.message.reply_text("✅ Твоё сообщение отправлено админам! Как только ответят, я тебе перешлю.")

async def admin_to_user(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.message.chat.id == CHAT_ID and update.message.reply_to_message:
        replied_msg_id = update.message.reply_to_message.message_id
        user_id = context.bot_data.get(replied_msg_id)
        if user_id:
            try:
                await context.bot.send_message(chat_id=user_id, text=f"💬 Ответ от администратора:\n\n{update.message.text}")
            except Forbidden:
                await update.message.reply_text(f"❌ Не удалось доставить ответ пользователю (ID: {user_id}). Он заблокировал бота.")
        else:
            await update.message.reply_text("❌ Не могу найти этого пользователя. Попроси его написать /start заново.")

if __name__ == "__main__":
    application = ApplicationBuilder().token(TOKEN).build()
    application.add_handler(CommandHandler("start", start))
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, user_to_admin))
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, admin_to_user))
    
    print("🚀 Бот успешно запущен и работает 24/7...")
    application.run_polling()
